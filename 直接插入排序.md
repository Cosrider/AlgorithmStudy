import random

def insertSort(arr):
    for i in range(len(arr)):
        j = i
        #使j之前的元素变得有序
        while j>0:
            if arr[j-1]>arr[j]:
                arr[j-1],arr[j] =arr[j],arr[j-1]
            j-=1

if __name__ =='__main__':
    arr = [random.randint(10,100) for x in range(10)]
    print(arr)
    insertSort(arr)
    print(arr)
